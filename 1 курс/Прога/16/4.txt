program project2;

var
  i, y, x: integer;
  f: file of integer;//файл типа интеджер
  name:string;

begin
  repeat
    writeln('MENU');
    writeln();
    writeln('1. Sozdat fail');
    writeln('2.chtenie faila');
    writeln('3.zavershit rabotu');
    writeln();
    write('deistvie: ');
    readln(y);
    case y of
     1: begin
      randomize;
      writeln();
      write('Vvedite imya faila: ');
      readln(name);
      assign(f, name);//связывание файловой переменной с файлом в текущем каталоге
      rewrite(f);//открытие файла для редактирования
      for i := 1 to 100 do//цикл, дающий x знач от -100 до 100
       begin
        x:= random(200) - 100;
        write(f, x);//записываем x в файл
       end;
      close(f);//закрытие файла
      writeln('chisla zapisani v fail');
      writeln();
     end;
     2: begin
      writeln;
      reset(f);//открытие файла для чтения
      while not EOF(f) do//пока не достигнут конец файла, делай
       begin
        read(f,x);//достаем х из файла
        writeln(x);//выводим х
       end;
      close(f);//хакрываем файл
      writeln;
     end;
    end;
  until y = 3;
end.
