program project3;

var
  i, y, x: integer;
  f, f1: file of integer;//файл типа интеджер
  name, name2:string;

begin
  repeat
    writeln('MENU');
    writeln();
    writeln('1. Sozdat fail 1');
    writeln('2. Sozdat fail 2 i perenesti na nego informaciu s faila 1');
    writeln('3. prochitat 1 fail');
    writeln('4. prochitat 2 fail');
    writeln('5.zavershit rabotu');
    writeln();
    write('deistvie: ');
    readln(y);
    case y of
     1: begin
      randomize;
      writeln();
      write('Vvedite imya faila: ');
      readln(name);
      assign(f, name);//связывание файловой переменной с файлом в текущем каталоге
      rewrite(f);//открытие файла для редактирования
      for i := 1 to 100 do//цикл, дающий x знач от -100 до 100
       begin
        x:= random(200) - 100;
        write(f, x);//записываем x в файл
       end;
      close(f);//закрытие файла
      writeln('chisla zapisani v fail');
      writeln();
     end;
     2: begin
      writeln;
      write('vvedite imya 2 faila:');
      readln(name2);
      reset(f);//открытие файла для чтения
      assign(f1, name2);
      rewrite(f1);
      while not EOF(f) do//пока не достигнут конец файла, делай
       begin
        read(f,x);//достаем х из файла
        write(f1, x);//выводим х
       end;
      close(f);//хакрываем файл
      close(f1);
      writeln('chisla skopirovany');
      writeln;
     end;
     3: begin
      writeln;
      reset(f);
      while not EOF(f) do
       begin
        read(f, x);
        writeln(x);
       end;
      close(f);
      writeln;
      end;
     4:begin
      writeln;
      reset(f1);
      while not EOF(f1) do
       begin
        read(f1, x);
        writeln(x);
       end;
      close(f1);
      writeln;
      end;
    end;
  until y = 5;
end.
